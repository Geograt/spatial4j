9   package org.locationtech.spatial4j.distance;
10  
11  import org.locationtech.spatial4j.shape.Point;
12  
13  /**
14   */
15  public abstract class AbstractDistanceCalculator implements DistanceCalculator {
16  
17    @Override
18    public double distance(Point from, Point to) {
19      return distance(from, to.getX(), to.getY());
20    }
21  
22    @Override
23    public boolean within(Point from, double toX, double toY, double distance) {
24      return distance(from, toX, toY) <= distance;
25    }
26  
27    @Override
28    public String toString() {
29      return getClass().getSimpleName();
30    }
31  }
