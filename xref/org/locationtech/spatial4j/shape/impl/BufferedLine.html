24  /**
25   * INTERNAL: A line between two points with a buffer distance extending in every direction. By
26   * contrast, an un-buffered line covers no area and as such is extremely unlikely to intersect with
27   * a point. BufferedLine isn't yet aware of geodesics (e.g. the anti-meridian); it operates in Euclidean
28   * space.
29   */
30  public class BufferedLine extends BaseShape<SpatialContext> {
31  
32    private final Point pA, pB;
33    private final double buf;
34    private final Rectangle bbox;
35    /**
36     * the primary line; passes through pA & pB
37     */
38    private final InfBufLine linePrimary;
39    /**
40     * perpendicular to the primary line, centered between pA & pB
41     */
42    private final InfBufLine linePerp;
43  
44    /**
45     * Creates a buffered line from pA to pB. The buffer extends on both sides of
46     * the line, making the width 2x the buffer. The buffer extends out from
47     * pA &amp; pB, making the line in effect 2x the buffer longer than pA to pB.
48     *
49     * @param pA  start point
50     * @param pB  end point
51     * @param buf the buffer distance in degrees
52     */
53    public BufferedLine(Point pA, Point pB, double buf, SpatialContext ctx) {
54      super(ctx);
55      assert buf >= 0;//TODO support buf=0 via another class ?
56  
57      /**
58       * If true, buf should bump-out from the pA & pB, in effect
59       *                  extending the line a little.
60       */
61      final boolean bufExtend = true;//TODO support false and make this a
62      // parameter
63  
64      this.pA = pA;
65      this.pB = pB;
66      this.buf = buf;
67  
68      double deltaY = pB.getY() - pA.getY();
69      double deltaX = pB.getX() - pA.getX();
70  
71      PointImpl center = new PointImpl(pA.getX() + deltaX / 2,
72          pA.getY() + deltaY / 2, null);
73  
74      double perpExtent = bufExtend ? buf : 0;
75  
76      if (deltaX == 0 && deltaY == 0) {
77        linePrimary = new InfBufLine(0, center, buf);
78        linePerp = new InfBufLine(Double.POSITIVE_INFINITY, center, buf);
79      } else {
80        linePrimary = new InfBufLine(deltaY / deltaX, center, buf);
81        double length = Math.sqrt(deltaX * deltaX + deltaY * deltaY);
82        linePerp = new InfBufLine(-deltaX / deltaY, center,
83            length / 2 + perpExtent);
84      }
85  
86      double minY, maxY;
87      double minX, maxX;
88      if (deltaX == 0) { // vertical
89        if (pA.getY() <= pB.getY()) {
90          minY = pA.getY();
91          maxY = pB.getY();
92        } else {
93          minY = pB.getY();
94          maxY = pA.getY();
95        }
96        minX = pA.getX() - buf;
97        maxX = pA.getX() + buf;
98        minY = minY - perpExtent;
99        maxY = maxY + perpExtent;
100 
101     } else {
102       if (!bufExtend) {
103         throw new UnsupportedOperationException("TODO");
104         //solve for B & A (C=buf), one is buf-x, other is buf-y.
105       }
106 
107       //Given a right triangle of A, B, C sides, C (hypotenuse) ==
108       // buf, and A + B == the bounding box offset from pA & pB in x & y.
109       double bboxBuf = buf * (1 + Math.abs(linePrimary.getSlope()))
110           * linePrimary.getDistDenomInv();
111       assert bboxBuf >= buf && bboxBuf <= buf * 1.5;
112 
113       if (pA.getX() <= pB.getX()) {
114         minX = pA.getX() - bboxBuf;
115         maxX = pB.getX() + bboxBuf;
116       } else {
117         minX = pB.getX() - bboxBuf;
118         maxX = pA.getX() + bboxBuf;
119       }
120       if (pA.getY() <= pB.getY()) {
121         minY = pA.getY() - bboxBuf;
122         maxY = pB.getY() + bboxBuf;
123       } else {
124         minY = pB.getY() - bboxBuf;
125         maxY = pA.getY() + bboxBuf;
126       }
127 
128     }
129     Rectangle bounds = ctx.getWorldBounds();
130 
131     bbox = ctx.makeRectangle(
132         Math.max(bounds.getMinX(), minX),
133         Math.min(bounds.getMaxX(), maxX),
134         Math.max(bounds.getMinY(), minY),
135         Math.min(bounds.getMaxY(), maxY));
136   }
137 
138   @Override
139   public boolean isEmpty() {
140     return pA.isEmpty();
141   }
142 
143   @Override
144   public Shape getBuffered(double distance, SpatialContext ctx) {
145     return new BufferedLine(pA, pB, buf + distance, ctx);
146   }

148   /**
149    * Calls {@link DistanceUtils#calcLonDegreesAtLat(double, double)} given pA or pB's latitude;
150    * whichever is farthest. It's useful to expand a buffer of a line segment when used in
151    * a geospatial context to cover the desired area.
152    */
153   public static double expandBufForLongitudeSkew(Point pA, Point pB,
154                                                  double buf) {
155     double absA = Math.abs(pA.getY());
156     double absB = Math.abs(pB.getY());
157     double maxLat = Math.max(absA, absB);
158     double newBuf = DistanceUtils.calcLonDegreesAtLat(maxLat, buf);
159 //    if (newBuf + maxLat >= 90) {
160 //      //TODO substitute spherical cap ?
161 //    }
162     assert newBuf >= buf;
163     return newBuf;
164   }

166   @Override
167   public SpatialRelation relate(Shape other) {
168     if (other instanceof Point)
169       return contains((Point) other) ? CONTAINS : DISJOINT;
170     if (other instanceof Rectangle)
171       return relate((Rectangle) other);
172     throw new UnsupportedOperationException();
173   }
174 
175   public SpatialRelation relate(Rectangle r) {
176     //Check BBox for disjoint & within.
177     SpatialRelation bboxR = bbox.relate(r);
178     if (bboxR == DISJOINT || bboxR == WITHIN)
179       return bboxR;
180     //Either CONTAINS, INTERSECTS, or DISJOINT
181 
182     Point scratch = new PointImpl(0, 0, null);
183     Point prC = r.getCenter();
184     SpatialRelation result = linePrimary.relate(r, prC, scratch);
185     if (result == DISJOINT)
186       return DISJOINT;
187     SpatialRelation resultOpp = linePerp.relate(r, prC, scratch);
188     if (resultOpp == DISJOINT)
189       return DISJOINT;
190     if (result == resultOpp)//either CONTAINS or INTERSECTS
191       return result;
192     return INTERSECTS;
193   }
194 
195   public boolean contains(Point p) {
196     //TODO check bbox 1st?
197     return linePrimary.contains(p) && linePerp.contains(p);
198   }

250   @Override
251   public boolean equals(Object o) {
252     if (this == o) return true;
253     if (o == null || getClass() != o.getClass()) return false;
254 
255     BufferedLine that = (BufferedLine) o;
256 
257     if (Double.compare(that.buf, buf) != 0) return false;
258     if (!pA.equals(that.pA)) return false;
259     if (!pB.equals(that.pB)) return false;
260 
261     return true;
262   }
263 
264   @Override
265   public int hashCode() {
266     int result;
267     long temp;
268     result = pA.hashCode();
269     result = 31 * result + pB.hashCode();
270     temp = buf != +0.0d ? Double.doubleToLongBits(buf) : 0L;
271     result = 31 * result + (int) (temp ^ (temp >>> 32));
272     return result;
273   }
